Switch case : 

Code reference  :

Example 1 : 

public static void main(String[] args) {

		String color = "Green";
		switch (color) {

		case "Red":
			System.out.println("STOP");
			break;
		case "Green":
			System.out.println("GO");
			break;
		case "Yellow":
			System.out.println("GET READY");
			break;

		default:
			System.out.println("Invalid Input");
		}

	}


Example 2 : 


public static void main(String[] args) {

		char grade = '@';
		switch (grade) {

		case 'A':
			System.out.println("Range -> 90 to 100  -> Excellent");
			break;
		case 'B':
			System.out.println("Range -> 80 to 90  -> Very Good");
			break;
		case 'C':
			System.out.println("Range -> 70 to 80  -> Good");
			break;
		case 'D':
			System.out.println("Range -> 60 to 70  -> Average");
			break;
		case 'F':
			System.out.println("Fail -> Better Luck Next Time .... ! ");
			break;
		default:
			System.out.println("Invalid Input");
		}

	}
